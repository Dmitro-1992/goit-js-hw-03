function makeArray(firstArray, secondArray, maxLength) {
  const result = [];

  for (let i = 0; i < firstArray.length; i++) {
    if (result.length === maxLength) break;
    result.push(firstArray[i]);
  }

  for (let i = 0; i < secondArray.length; i++) {
    if (result.length === maxLength) break;
    result.push(secondArray[i]);
  }

  return result;
}
